Acwing算法基础课

基础算法

排序

快排

$$
O(nlogn)
$$

快排算法基于分治策略

  1. 确定分界点x

    取左边界、右边界、中间值、随机值都可以

  2. 调整区间

    划分为两个区间（分界点不一定是x）

    小于等于x的数在左区间

    大于等于x的数在右区间

  3. 递归的处理左右两区间

模板：

void quick_sort(int q[], int l,int r)
{
    int i = l - 1, j = r + 1;
    int x = q[l];
    while(i < j)
    {
        do i++; while(q[i] < x);
        do j--; while(q[j] > x);
        if(i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1;r);
}

归并排序

$$
O(nlogn)
$$

归并排序算法也是基于分治策略

  1. 以中间为分界点 mid = (l+r)/2
  2. 递归排序左区间右区间
  3. 归并两个有序区间

模板：

void merge_sort(int q[], int l, int r)
{
    if(l >= r) return;
    int mid = l + r >> 1;
    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);
    
    int k = 0, i = l, j = mid + 1;
    while(i <= mid && j <= r)

    	else tmp[k++] = q[j++];
    while(i <= mid) tmp[k++] = q[i++];
    while(j <= r) tmp[k++] = q[j++];
    
    for(i = l, j = 0;i <= r;i++, j++)
        q[i] = tmp[j];
}

快排是不稳定的，归并是稳定的

整数

int bsearch_1(int l, int r)
{
    while(l < r)
    {
        int mid = l+r >> 1;
        if(check(mid)) r = mid;
        else l = mid + 1;
        
    }
    return 1;
}
int bsearch_2(int l, int r)
{
    while(l < r)
    {
        int mid = l + r + 1 >>1;
        if(check(mid)) l = mid;
        else r = mid -1;
    }
    return 1;
}

浮点数

浮点数二分无需处理边界

double bsearch_3(int l, int r)
{
    const double eps = 1e-6; //精度要求
    while(r-1>eps)
    {
        double mid = (l+r)/2;
        if(check(mid)) r = mid;
        else l=mid;
    }
    return 1;
}

高精度

用数组存储大整数，个位先存，即存储在数组下标为0的位置，十位存在1，以此类推

高精度加法

vector<int> add(vector<int> &A, vector<int> &B)
{
    vector<int> C;
    int t;
    for(int i=0;i<A.size() || i<B.size();i++)
    {
        if(i<A.size()) t+=A[i];
        if(i<B.size()) t+=B[i];
        C.push_back(t%10);
        t/=10;
    }
    if(t) C.push_back(1);
    return C;
}

高精度减法

vector<int>  sub(vector<int> &A, vector<int> &B)
{
    vector<int> C;
    for(int i=0,t=0;i<A.size();i++)
    {
        t = A[i]-t;
        if(i<B.size()) t-=B[i];
        C.push_back((t+10)%10);
        if(t<0) t=1;
        else t=0;
    }
    while(C.size()>1 && C.back()==0) C.pop_back();
    return C;
}

高精度乘法

vector<int> mul(vector<int> &A, int b)
{
    vector<int> C;
    int t=0;
    for(int i=0;i<A.size() || t;i++)
    {
        if(i < A.size()) t+=A[i]*b;
        C.push_back(t%10);
        t/=10;
    }
    return C;
    
}

高精度除法

vector<int> div(vector<int> &A, int b, int &r)
{
    vector<int> C;
    r=0;
    for(int i=0;i=A.size()-1;i>=0;i--)
    {
        r = r*10 + A[i];
        C.push_back(r / b);
        r%=b;
    }
    reverse(C.begin(), C.end());
    while(C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
    
}
