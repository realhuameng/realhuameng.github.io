    排序

    快排

    题目链接：785.快速排序

    快排算法基于分治策略

    1. 确定分界点x

      取左边界、右边界、中间值、随机值都可以

    2. 调整区间

      划分为两个区间（分界点不一定是x）

      小于等于x的数在左区间

      大于等于x的数在右区间

    3. 递归的处理左右两区间

    模板：

    void quick_sort(int q[], int l,int r)
{
    int i = l - 1, j = r + 1;
    int x = q[l];
    while(i < j)
    {
        do i++; while(q[i] < x);
        do j--; while(q[j] > x);
        if(i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1;r);
}

    归并排序

    题目链接：787.归并排序
    $$
    O(nlogn)
    $$

    归并排序算法也是基于分治策略

    1. 以中间为分界点 mid = (l+r)/2
    2. 递归排序左区间右区间
    3. 归并两个有序区间

    模板：

    void merge_sort(int q[], int l, int r)
{
    if(l >= r) return;
    int mid = l + r >> 1;
    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);
    
    int k = 0, i = l, j = mid + 1;
    while(i <= mid && j <= r)
        if(q[i] <= q[j]) tmp[k++] = q[i++];
    	else tmp[k++] = q[j++];
    while(i <= mid) tmp[k++] = q[i++];
    while(j <= r) tmp[k++] = q[j++];
    
    for(i = l, j = 0;i <= r;i++, j++)
        q[i] = tmp[j];
}

    快排是不稳定的，归并是稳定的

    高精度

    用数组存储大整数，个位先存，即存储在数组下标为0的位置，十位存在1，以此类推

    高精度加法

    vector<int> add(vector<int> &A, vector<int> &B)
{
    vector<int> C;
    int t;
    for(int i=0;i<A.size() || i<B.size();i++)
    {
        if(i<A.size()) t+=A[i];
        if(i<B.size()) t+=B[i];
        C.push_back(t%10);
        t/=10;
    }
    if(t) C.push_back(1);
    return C;
}

    高精度减法

    vector<int>  sub(vector<int> &A, vector<int> &B)
{
    vector<int> C;
    for(int i=0,t=0;i<A.size();i++)
    {
        t = A[i]-t;
        if(i<B.size()) t-=B[i];
        C.push_back((t+10)%10);
        if(t<0) t=1;
        else t=0;
    }
    while(C.size()>1 && C.back()==0) C.pop_back();
    return C;
}

    高精度乘法

    vector<int> mul(vector<int> &A, int b)
{
    vector<int> C;
    int t=0;
    for(int i=0;i<A.size() || t;i++)
    {
        if(i < A.size()) t+=A[i]*b;
        C.push_back(t%10);
        t/=10;
    }
    return C;
    
}

    高精度除法

    vector<int> div(vector<int> &A, int b, int &r)
{
    vector<int> C;
    r=0;
    for(int i=0;i=A.size()-1;i--)
    {
        r = r*10 + A[i];
        C.push_back(r / b);
        r%=b;
    }
    reverse(C.begin(), C.end());
    while(C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
    
}

    双链表

    int e[N], l[N], r[N], idx;
//初始化
void init(){
    //0是左端点，1是右端点
    r[0] = 1;
    l[1] = 0;
    idx = 2;
}
//在第k个节点后插入x
void add(int k, int x){
    e[idx] = x
    r[idx] = r[k];
    l[idx] = k;
    l[r[k]] = idx;
    r[k] = idx;
    
}
//删除
void remove(int k){
    r[l[k]] = r[k];
    l[r[k]] = l[k];
}

int n, m;
int p[N], s[M];
int next[N];
int main(){
    //求next数组
    for(int i=2, j=0;i <=n;i++)
    {
        while(j && p[i] != p[j+1]) j = next[j];
         if(p[i] == p[j+1]) j++;
         next[i] = j;
}
    
    //KMP匹配过程
    for(int i=1, j=0;i<=m;i++)
    {
        //j没退回起点且当前的s[i]无法匹配
        while(j && s[i] != p[j+1]) j = next[j];
