    快排

    题目链接：785.快速排序

    快排算法基于分治策略

    1. 确定分界点x

      取左边界、右边界、中间值、随机值都可以

    2. 调整区间

      划分为两个区间（分界点不一定是x）

      小于等于x的数在左区间

      大于等于x的数在右区间

    3. 递归的处理左右两区间

    模板：

    void quick_sort(int q[], int l,int r)
{
    int i = l - 1, j = r + 1;
    int x = q[l];
    while(i < j)
    {
        do i++; while(q[i] < x);
        do j--; while(q[j] > x);
        if(i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1;r);
}

    归并排序

    题目链接：787.归并排序
    $$
    O(nlogn)
    $$

    归并排序算法也是基于分治策略

    1. 以中间为分界点 mid = (l+r)/2
    2. 递归排序左区间右区间
    3. 归并两个有序区间

    模板：

    void merge_sort(int q[], int l, int r)
{
    if(l >= r) return;
    int mid = l + r >> 1;
    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);
    
    int k = 0, i = l, j = mid + 1;
    while(i <= mid && j <= r)
        if(q[i] <= q[j]) tmp[k++] = q[i++];
    	else tmp[k++] = q[j++];
    while(i <= mid) tmp[k++] = q[i++];
    while(j <= r) tmp[k++] = q[j++];
    
    for(i = l, j = 0;i <= r;i++, j++)
        q[i] = tmp[j];
}

    堆

    以小根堆为例

    heap数组下标从1开始

    1.插入一个数 heap[++size] = x; up(size);

    2.求集合中最小值 heap[1]；

    3.删除最小值 heap[1] = heap[size]; size--; down(1);

    4.删除任意一个元素 heap[k] = heap[size]; size--; down(k); up(k);

    5.修改任意一个元素 heap[k] = x; down(k); up(k);

    6.建堆：for(int i = n/2;i;i--) down(i);

    存储：x的左儿子是2x，右儿子是2x+1

    最短路径问题

    朴素Dijkstra

    1. dist[1] = 0, dist[i] = ∞，s: 当前已确定最短距离的点

    2. for(i = 0~n)

      t<-不在s中的距离最近的点

      s<-t

      用t来更新其他点的距离

    int g[N][N]; //稠密图
int dist[N];
bool st[N];
int dijkstra()
{
    memset(dist, 0x3f, sizeof dist);
    dist[1]=0;
    for(int i=0;i<n;i++)
    {
        int t=-1;
        for(int j=1;j<=n;j++)
            if(!st[j] && (t==-1 || dist[t] > dist[j]))
                t=j;
        for(int j=1;j<n;j++)
        {
            dist[j] = min(dist[j], dist[t] + g[t][j]);

        }
        st[t] = true;
    }
    if(dist[n]==0x3f3f3f3f) return -1;
    return dist[n];
}

typedef pair<int, int> PII;

void add(int a, int b, int c)
{
    e[idx] = b;
    w[idx] = c;

    h[a] = idx++;
}
int dijkstra()
{
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;
    priority_queue<PII, vector<PII>, greater<PII>> heap;
    heap.push({0, 1});
    
    while(heap.size())
    {
        auto t = heap.top();
        int ver = t.second, distance=t.first;
        if(st[ver]) continue;
        st[ver] = true;
        for(int i=h[ver],i!=-1;i=ne[i])
        {
            int j=e[i];
            if(dist[j] > distance + w[i])
            {
                dist[j] = distance + w[i];
                heap.push({dist[j], j});
            }
        }

    }
    if(dist[n] == 0x3f3f3f3f) return -1;
    else return dist[n];
}

        for(int i=h[t];i!=-1;i=ne[i])
        {
            int j=e[i];
            if(dist[j] > dist[t]+w[i])
            {
                dist[j] = dist[t]+w[i];
                if(!st[j])
                {
                    q.push(j);
                    st[j]=false;
                }
            }
        }
    }
}

                	t=j;
