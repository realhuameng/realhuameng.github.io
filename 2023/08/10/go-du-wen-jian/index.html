Go_读文件

错误检查

func check(e error){
    if e != nil{
        panic(e)
    }
}

示例

func main(){
    dat,err := os.ReadFile("/tmp/dat")
    check(err)
    fmt.Print(string(dat))
    
    f,err := os.Open("/tmp/dat") //获取一个os.File值
    check(err)
    
    b1 := make([]byte,5)
    n1,err := f.Read(b1)
    check(err)
    fmt.Println("%d bytes: %s\n",n1,string(b1[:n1]))
}
